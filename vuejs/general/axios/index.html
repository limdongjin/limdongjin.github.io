var qs = require('qs')

this.axios.post(`${baseURI}/posts`,
    qs.stringify({
        title: this.title,
        content: this.content,
        category: this.category
    }),
    {
        headers: {'Content-Type': 'application/x-www-form-urlencoded'}
    })
    .then((result) => {
            console.log(result)
